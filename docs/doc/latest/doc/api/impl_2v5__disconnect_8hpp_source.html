65inline
66std::vector<as::const_buffer> disconnect_packet::const_buffer_sequence() const {
67 std::vector<as::const_buffer> ret;
68 ret.reserve(num_of_const_buffer_sequence());
69 ret.emplace_back(as::buffer(&fixed_header_, 1));
70 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
71
72 if (reason_code_) {
73 ret.emplace_back(as::buffer(&*reason_code_, 1));
74
75 if (property_length_buf_.size() != 0) {
76 ret.emplace_back(as::buffer(property_length_buf_.data(), property_length_buf_.size()));
77 auto props_cbs = async_mqtt::const_buffer_sequence(props_);
78 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));
79 }
80 }
81
82 return ret;
83}

93inline
94std::size_t disconnect_packet::num_of_const_buffer_sequence() const {
95 return
96 1 + // fixed header
97 1 + // remaining length
98 [&] () -> std::size_t {
99 if (reason_code_) {
100 return
101 1 + // reason_code
102 [&] () -> std::size_t {
103 if (property_length_buf_.size() == 0) return 0;
104 return
105 1 + // property length
106 async_mqtt::num_of_const_buffer_sequence(props_);
107 }();
108 }
109 return 0;
110 }();
111}

124inline
125disconnect_packet::disconnect_packet(
126 std::optional<disconnect_reason_code> reason_code,
127 properties props
128)
129 : fixed_header_{
130 make_fixed_header(control_packet_type::disconnect, 0b0000)
131 },
132 remaining_length_{
133 0
134 },
135 reason_code_{reason_code},
136 property_length_(async_mqtt::size(props)),
137 props_(force_move(props))
138{
139 using namespace std::literals;
140
141 auto guard = unique_scope_guard(
142 [&] {
143 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
144 for (auto e : rb) {
145 remaining_length_buf_.push_back(e);
146 }
147 }
148 );
149
150 if (!reason_code_) return;
151 remaining_length_ += 1;
152
153 if (property_length_ == 0) return;
154
155 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
156 for (auto e : pb) {
157 property_length_buf_.push_back(e);
158 }
159
160 for (auto const& prop : props_) {
161 auto id = prop.id();
162 if (!validate_property(property_location::disconnect, id)) {
163 throw make_error(
164 errc::bad_message,
165 "v5::disconnect_packet property "s + id_to_str(id) + " is not allowed"
166 );
167 }
168 }
169
170 remaining_length_ += property_length_buf_.size() + property_length_;
171}

173// private constructor for internal use
174inline
175disconnect_packet::disconnect_packet(buffer buf) {
176 // fixed_header
177 if (buf.empty()) {
178 throw make_error(
179 errc::bad_message,
180 "v5::disconnect_packet fixed_header doesn't exist"
181 );
182 }
183 fixed_header_ = static_cast<std::uint8_t>(buf.front());
184 buf.remove_prefix(1);
185
186 // remaining_length
187 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
188 remaining_length_ = *vl_opt;
189 }
190 else {
191 throw make_error(errc::bad_message, "v5::disconnect_packet remaining length is invalid");
192 }
193
194 if (remaining_length_ == 0) {
195 if (!buf.empty()) {
196 throw make_error(errc::bad_message, "v5::disconnect_packet remaining length is invalid");
197 }
198 return;
199 }
200
201 // reason_code
202 reason_code_.emplace(static_cast<disconnect_reason_code>(buf.front()));
203 buf.remove_prefix(1);
204 switch (*reason_code_) {
205 case disconnect_reason_code::normal_disconnection:
206 case disconnect_reason_code::disconnect_with_will_message:
207 case disconnect_reason_code::unspecified_error:
208 case disconnect_reason_code::malformed_packet:
209 case disconnect_reason_code::protocol_error:
210 case disconnect_reason_code::implementation_specific_error:
211 case disconnect_reason_code::not_authorized:
212 case disconnect_reason_code::server_busy:
213 case disconnect_reason_code::server_shutting_down:
214 case disconnect_reason_code::keep_alive_timeout:
215 case disconnect_reason_code::session_taken_over:
216 case disconnect_reason_code::topic_filter_invalid:
217 case disconnect_reason_code::topic_name_invalid:
218 case disconnect_reason_code::receive_maximum_exceeded:
219 case disconnect_reason_code::topic_alias_invalid:
220 case disconnect_reason_code::packet_too_large:
221 case disconnect_reason_code::message_rate_too_high:
222 case disconnect_reason_code::quota_exceeded:
223 case disconnect_reason_code::administrative_action:
224 case disconnect_reason_code::payload_format_invalid:
225 case disconnect_reason_code::retain_not_supported:
226 case disconnect_reason_code::qos_not_supported:
227 case disconnect_reason_code::use_another_server:
228 case disconnect_reason_code::server_moved:
229 case disconnect_reason_code::shared_subscriptions_not_supported:
230 case disconnect_reason_code::connection_rate_exceeded:
231 case disconnect_reason_code::maximum_connect_time:
232 case disconnect_reason_code::subscription_identifiers_not_supported:
233 case disconnect_reason_code::wildcard_subscriptions_not_supported:
234 break;
235 default:
236 throw make_error(
237 errc::bad_message,
238 "v5::disconnect_packet connect reason_code is invalid"
239 );
240 break;
241 }
242
243 if (remaining_length_ == 1) {
244 if (!buf.empty()) {
245 throw make_error(errc::bad_message, "v5::disconnect_packet remaining length is invalid");
246 }
247 return;
248 }
249
250 // property
251 auto it = buf.begin();
252 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
253 property_length_ = *pl_opt;
254 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
255 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
256 if (buf.size() < property_length_) {
257 throw make_error(
258 errc::bad_message,
259 "v5::disconnect_packet properties_don't match its length"
260 );
261 }
262 auto prop_buf = buf.substr(0, property_length_);
263 props_ = make_properties(prop_buf, property_location::disconnect);
264 buf.remove_prefix(property_length_);
265 }
266 else {
267 throw make_error(
268 errc::bad_message,
269 "v5::disconnect_packet property_length is invalid"
270 );
271 }
272
273 if (!buf.empty()) {
274 throw make_error(
275 errc::bad_message,
276 "v5::disconnect_packet properties don't match its length"
277 );
278 }
279}
