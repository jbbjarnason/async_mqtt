47inline
48connect_packet::connect_packet(
49 bool clean_session,
50 std::uint16_t keep_alive_sec,
51 std::string client_id,
52 std::optional<will> w,
53 std::optional<std::string> user_name,
54 std::optional<std::string> password
55)
56 : fixed_header_{
57 make_fixed_header(control_packet_type::connect, 0b0000)
58 },
59 connect_flags_{0},
60 // protocol name length, protocol name, protocol level, connect flag, client id length, client id, keep alive
61 remaining_length_(
62 2 + // protocol name length
63 4 + // protocol name
64 1 + // protocol level
65 1 + // connect flag
66 2 + // keep alive
67 2 + // client id length
68 client_id.size() // client id
69 ),
70 protocol_name_and_level_{0x00, 0x04, 'M', 'Q', 'T', 'T', 0x04},
71 client_id_{force_move(client_id)},
72 client_id_length_buf_(2),
73 keep_alive_buf_(2)
74{
75 endian_store(keep_alive_sec, keep_alive_buf_.data());
76 endian_store(boost::numeric_cast<std::uint16_t>(client_id_.size()), client_id_length_buf_.data());
77
78 if (!utf8string_check(client_id_)) {
79 throw make_error(
80 errc::bad_message,
81 "v3_1_1::connect_packet client_id invalid utf8"
82 );
83 }
84
85 if (clean_session) connect_flags_ |= connect_flags::mask_clean_session;
86 if (user_name) {
87 if (!utf8string_check(*user_name)) {
88 throw make_error(
89 errc::bad_message,
90 "v3_1_1::connect_packet user name invalid utf8"
91 );
92 }
93 connect_flags_ |= connect_flags::mask_user_name_flag;
94 user_name_ = buffer{force_move(*user_name)};
95 user_name_length_buf_ = endian_static_vector(boost::numeric_cast<std::uint16_t>(user_name_.size()));
96 remaining_length_ += 2 + user_name_.size();
97 }
98 if (password) {
99 connect_flags_ |= connect_flags::mask_password_flag;
100 password_ = buffer{force_move(*password)};
101 password_length_buf_ = endian_static_vector(boost::numeric_cast<std::uint16_t>(password_.size()));
102 remaining_length_ += 2 + password_.size();
103 }
104 if (w) {
105 connect_flags_ |= connect_flags::mask_will_flag;
106 if (w->get_retain() == pub::retain::yes) connect_flags_ |= connect_flags::mask_will_retain;
107 connect_flags::set_will_qos(connect_flags_, w->get_qos());
108 if (!utf8string_check(w->topic())) {
109 throw make_error(
110 errc::bad_message,
111 "v3_1_1::connect_packet will topic invalid utf8"
112 );
113 }
114 will_topic_ = force_move(w->topic_as_buffer());
115 will_topic_length_buf_ = endian_static_vector(boost::numeric_cast<std::uint16_t>(will_topic_.size()));
116 if (w->message().size() > 0xffffL) {
117 throw make_error(
118 errc::bad_message,
119 "v3_1_1::connect_packet will message too long"
120 );
121 }
122 will_message_ = force_move(w->message_as_buffer());
123 will_message_length_buf_ = endian_static_vector(boost::numeric_cast<std::uint16_t>(will_message_.size()));
124
125 remaining_length_ += 2 + will_topic_.size() + 2 + will_message_.size();
126 }
127
128 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
129 for (auto e : rb) {
130 remaining_length_buf_.push_back(e);
131 }
132}

139inline
140std::vector<as::const_buffer> connect_packet::const_buffer_sequence() const {
141 std::vector<as::const_buffer> ret;
142 ret.reserve(num_of_const_buffer_sequence());
143
144 ret.emplace_back(as::buffer(&fixed_header_, 1));
145 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
146 ret.emplace_back(as::buffer(protocol_name_and_level_.data(), protocol_name_and_level_.size()));
147 ret.emplace_back(as::buffer(&connect_flags_, 1));
148 ret.emplace_back(as::buffer(keep_alive_buf_.data(), keep_alive_buf_.size()));
149
150 ret.emplace_back(as::buffer(client_id_length_buf_.data(), client_id_length_buf_.size()));
151 ret.emplace_back(as::buffer(client_id_));
152
153 if (connect_flags::has_will_flag(connect_flags_)) {
154 ret.emplace_back(as::buffer(will_topic_length_buf_.data(), will_topic_length_buf_.size()));
155 ret.emplace_back(as::buffer(will_topic_));
156 ret.emplace_back(as::buffer(will_message_length_buf_.data(), will_message_length_buf_.size()));
157 ret.emplace_back(as::buffer(will_message_));
158 }
159
160 if (connect_flags::has_user_name_flag(connect_flags_)) {
161 ret.emplace_back(as::buffer(user_name_length_buf_.data(), user_name_length_buf_.size()));
162 ret.emplace_back(as::buffer(user_name_));
163 }
164
165 if (connect_flags::has_password_flag(connect_flags_)) {
166 ret.emplace_back(as::buffer(password_length_buf_.data(), password_length_buf_.size()));
167 ret.emplace_back(as::buffer(password_));
168 }
169
170 return ret;
171}

181inline
182std::size_t connect_packet::num_of_const_buffer_sequence() const {
183 return
184 1 + // fixed header
185 1 + // remaining length
186 1 + // protocol name and level
187 1 + // connect flags
188 1 + // keep alive
189
190 2 + // client id length, client id
191
192 [&] () -> std::size_t {
193 if (connect_flags::has_will_flag(connect_flags_)) {
194 return
195 2 + // will topic name length, will topic name
196 2; // will message length, will message
197 }
198 return 0;
199 } () +
200 [&] () -> std::size_t {
201 if (connect_flags::has_user_name_flag(connect_flags_)) {
202 return 2; // user name length, user name
203 }
204 return 0;
205 } () +
206 [&] () -> std::size_t {
207 if (connect_flags::has_password_flag(connect_flags_)) {
208 return 2; // password length, password
209 }
210 return 0;
211 } ();
212}

268inline
269connect_packet::connect_packet(buffer buf) {
270 // fixed_header
271 if (buf.empty()) {
272 throw make_error(
273 errc::bad_message,
274 "v3_1_1::connect_packet fixed_header doesn't exist"
275 );
276 }
277 fixed_header_ = static_cast<std::uint8_t>(buf.front());
278 buf.remove_prefix(1);
279 auto cpt_opt = get_control_packet_type_with_check(fixed_header_);
280 if (!cpt_opt || *cpt_opt != control_packet_type::connect) {
281 throw make_error(
282 errc::bad_message,
283 "v3_1_1::connect_packet fixed_header is invalid"
284 );
285 }
286
287 // remaining_length
288 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
289 remaining_length_ = *vl_opt;
290 }
291 else {
292 throw make_error(errc::bad_message, "v3_1_1::connect_packet remaining length is invalid");
293 }
294 if (remaining_length_ != buf.size()) {
295 throw make_error(errc::bad_message, "v3_1_1::connect_packet remaining length doesn't match buf.size()");
296 }
297
298 // protocol name and level
299 if (!insert_advance(buf, protocol_name_and_level_)) {
300 throw make_error(
301 errc::bad_message,
302 "v3_1_1::connect_packet length of protocol_name or level is invalid"
303 );
304 }
305 static_vector<char, 7> expected_protocol_name_and_level {
306 0, 4, 'M', 'Q', 'T', 'T', 4
307 };
308 if (protocol_name_and_level_ != expected_protocol_name_and_level) {
309 throw make_error(
310 errc::bad_message,
311 "v3_1_1::connect_packet contents of protocol_name or level is invalid"
312 );
313 }
314
315 // connect_flags
316 if (buf.size() < 1) {
317 throw make_error(
318 errc::bad_message,
319 "v3_1_1::connect_packet connect_flags doesn't exist"
320 );
321 }
322 connect_flags_ = buf.front();
323 if (connect_flags_ & 0b00000001) {
324 throw make_error(
325 errc::bad_message,
326 "v3_1_1::connect_packet connect_flags reserved bit0 is 1 (must be 0)"
327 );
328 }
329 buf.remove_prefix(1);
330
331 // keep_alive
332 if (!insert_advance(buf, keep_alive_buf_)) {
333 throw make_error(
334 errc::bad_message,
335 "v3_1_1::connect_packet keep_alive is invalid"
336 );
337 }
338
339 // client_id_length
340 if (!insert_advance(buf, client_id_length_buf_)) {
341 throw make_error(
342 errc::bad_message,
343 "v3_1_1::connect_packet length of client_id is invalid"
344 );
345 }
346 auto client_id_length = endian_load<std::uint16_t>(client_id_length_buf_.data());
347
348 // client_id
349 if (buf.size() < client_id_length) {
350 throw make_error(
351 errc::bad_message,
352 "v3_1_1::connect_packet client_id doesn't match its length"
353 );
354 }
355 client_id_ = buf.substr(0, client_id_length);
356 if (!utf8string_check(client_id_)) {
357 throw make_error(
358 errc::bad_message,
359 "v3_1_1::connect_packet client_id invalid utf8"
360 );
361 }
362 buf.remove_prefix(client_id_length);
363
364 // will
365 if (connect_flags::has_will_flag(connect_flags_)) {
366 auto will_qos = connect_flags::will_qos(connect_flags_);
367 if (will_qos != qos::at_most_once &&
368 will_qos != qos::at_least_once &&
369 will_qos != qos::exactly_once) {
370 throw make_error(
371 errc::bad_message,
372 "v3_1_1::connect_packet will_qos is invalid"
373 );
374 }
375 // will_topic_length
376 if (!insert_advance(buf, will_topic_length_buf_)) {
377 throw make_error(
378 errc::bad_message,
379 "v3_1_1::connect_packet length of will_topic is invalid"
380 );
381 }
382 auto will_topic_length = endian_load<std::uint16_t>(will_topic_length_buf_.data());
383
384 // will_topic
385 if (buf.size() < will_topic_length) {
386 throw make_error(
387 errc::bad_message,
388 "v3_1_1::connect_packet will_topic doesn't match its length"
389 );
390 }
391 will_topic_ = buf.substr(0, will_topic_length);
392 if (!utf8string_check(will_topic_)) {
393 throw make_error(
394 errc::bad_message,
395 "v3_1_1::connect_packet will topic invalid utf8"
396 );
397 }
398 buf.remove_prefix(will_topic_length);
399
400 // will_message_length
401 if (!insert_advance(buf, will_message_length_buf_)) {
402 throw make_error(
403 errc::bad_message,
404 "v3_1_1::connect_packet length of will_message is invalid"
405 );
406 }
407 auto will_message_length = endian_load<std::uint16_t>(will_message_length_buf_.data());
408
409 // will_message
410 if (buf.size() < will_message_length) {
411 throw make_error(
412 errc::bad_message,
413 "v3_1_1::connect_packet will_message doesn't match its length"
414 );
415 }
416 will_message_ = buf.substr(0, will_message_length);
417 buf.remove_prefix(will_message_length);
418 }
419 else {
420 auto will_retain = connect_flags::will_retain(connect_flags_);
421 auto will_qos = connect_flags::will_qos(connect_flags_);
422 if (will_retain == pub::retain::yes) {
423 throw make_error(
424 errc::bad_message,
425 "v3_1_1::connect_packet combination of will_flag and will_retain is invalid"
426 );
427 }
428 if (will_qos != qos::at_most_once) {
429 throw make_error(
430 errc::bad_message,
431 "v3_1_1::connect_packet combination of will_flag and will_qos is invalid"
432 );
433 }
434 }
435 // user_name
436 if (connect_flags::has_user_name_flag(connect_flags_)) {
437 // user_name_topic_name_length
438 if (!insert_advance(buf, user_name_length_buf_)) {
439 throw make_error(
440 errc::bad_message,
441 "v3_1_1::connect_packet length of user_name is invalid"
442 );
443 }
444 auto user_name_length = endian_load<std::uint16_t>(user_name_length_buf_.data());
445
446 // user_name
447 if (buf.size() < user_name_length) {
448 throw make_error(
449 errc::bad_message,
450 "v3_1_1::connect_packet user_name doesn't match its length"
451 );
452 }
453 user_name_ = buf.substr(0, user_name_length);
454 if (!utf8string_check(user_name_)) {
455 throw make_error(
456 errc::bad_message,
457 "v3_1_1::connect_packet user name invalid utf8"
458 );
459 }
460 buf.remove_prefix(user_name_length);
461 }
462
463 // password
464 if (connect_flags::has_password_flag(connect_flags_)) {
465 // password_topic_name_length
466 if (!insert_advance(buf, password_length_buf_)) {
467 throw make_error(
468 errc::bad_message,
469 "v3_1_1::connect_packet length of password is invalid"
470 );
471 }
472 auto password_length = endian_load<std::uint16_t>(password_length_buf_.data());
473
474 // password
475 if (buf.size() != password_length) {
476 throw make_error(
477 errc::bad_message,
478 "v3_1_1::connect_packet password doesn't match its length"
479 );
480 }
481 password_ = buf.substr(0, password_length);
482 buf.remove_prefix(password_length);
483 }
484}
