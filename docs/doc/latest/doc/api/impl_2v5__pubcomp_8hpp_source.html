76template <std::size_t PacketIdBytes>
77inline
78std::vector<as::const_buffer> basic_pubcomp_packet<PacketIdBytes>::const_buffer_sequence() const {
79 std::vector<as::const_buffer> ret;
80 ret.reserve(num_of_const_buffer_sequence());
81 ret.emplace_back(as::buffer(&fixed_header_, 1));
82 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
83
84 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
85
86 if (reason_code_) {
87 ret.emplace_back(as::buffer(&*reason_code_, 1));
88
89 if (property_length_buf_.size() != 0) {
90 ret.emplace_back(as::buffer(property_length_buf_.data(), property_length_buf_.size()));
91 auto props_cbs = async_mqtt::const_buffer_sequence(props_);
92 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));
93 }
94 }
95
96 return ret;
97}

109inline
110constexpr std::size_t basic_pubcomp_packet<PacketIdBytes>::num_of_const_buffer_sequence() const {
111 return
112 1 + // fixed header
113 1 + // remaining length
114 1 + // packet_id
115 [&] () -> std::size_t {
116 if (reason_code_) {
117 return
118 1 + // reason_code
119 [&] () -> std::size_t {
120 if (property_length_buf_.size() == 0) return 0;
121 return
122 1 + // property length
123 async_mqtt::num_of_const_buffer_sequence(props_);
124 }();
125 }
126 return 0;
127 }();

149template <std::size_t PacketIdBytes>
150inline
151basic_pubcomp_packet<PacketIdBytes>::basic_pubcomp_packet(
152 typename basic_packet_id_type<PacketIdBytes>::type packet_id,
153 std::optional<pubcomp_reason_code> reason_code,
154 properties props
155)
156 : fixed_header_{
157 make_fixed_header(control_packet_type::pubcomp, 0b0000)
158 },
159 remaining_length_{
160 PacketIdBytes
161 },
162 packet_id_(packet_id_.capacity()),
163 reason_code_{reason_code},
164 property_length_(async_mqtt::size(props)),
165 props_(force_move(props))
166{
167 using namespace std::literals;
168 endian_store(packet_id, packet_id_.data());
169
170 auto guard = unique_scope_guard(
171 [&] {
172 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
173 for (auto e : rb) {
174 remaining_length_buf_.push_back(e);
175 }
176 }
177 );
178
179 if (!reason_code_) return;
180 remaining_length_ += 1;
181
182 if (property_length_ == 0) return;
183
184 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
185 for (auto e : pb) {
186 property_length_buf_.push_back(e);
187 }
188
189 for (auto const& prop : props_) {
190 auto id = prop.id();
191 if (!validate_property(property_location::pubcomp, id)) {
192 throw make_error(
193 errc::bad_message,
194 "v5::pubcomp_packet property "s + id_to_str(id) + " is not allowed"
195 );
196 }
197 }
198
199 remaining_length_ += property_length_buf_.size() + property_length_;
200}

202template <std::size_t PacketIdBytes>
203inline
204basic_pubcomp_packet<PacketIdBytes>::basic_pubcomp_packet(buffer buf) {
205 // fixed_header
206 if (buf.empty()) {
207 throw make_error(
208 errc::bad_message,
209 "v5::pubcomp_packet fixed_header doesn't exist"
210 );
211 }
212 fixed_header_ = static_cast<std::uint8_t>(buf.front());
213 buf.remove_prefix(1);
214
215 // remaining_length
216 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
217 remaining_length_ = *vl_opt;
218 }
219 else {
220 throw make_error(errc::bad_message, "v5::pubcomp_packet remaining length is invalid");
221 }
222
223 // packet_id
224 if (!insert_advance(buf, packet_id_)) {
225 throw make_error(
226 errc::bad_message,
227 "v5::pubcomp_packet packet_id doesn't exist"
228 );
229 }
230
231 if (remaining_length_ == PacketIdBytes) {
232 if (!buf.empty()) {
233 throw make_error(errc::bad_message, "v5::pubcomp_packet remaining length is invalid");
234 }
235 return;
236 }
237
238 // reason_code
239 reason_code_.emplace(static_cast<pubcomp_reason_code>(buf.front()));
240 buf.remove_prefix(1);
241 switch (*reason_code_) {
242 case pubcomp_reason_code::success:
243 case pubcomp_reason_code::packet_identifier_not_found:
244 break;
245 default:
246 throw make_error(
247 errc::bad_message,
248 "v5::pubcomp_packet connect reason_code is invalid"
249 );
250 break;
251 }
252
253 if (remaining_length_ == 3) {
254 if (!buf.empty()) {
255 throw make_error(errc::bad_message, "v5::pubcomp_packet remaining length is invalid");
256 }
257 return;
258 }
259
260 // property
261 auto it = buf.begin();
262 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
263 property_length_ = *pl_opt;
264 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
265 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
266 if (buf.size() < property_length_) {
267 throw make_error(
268 errc::bad_message,
269 "v5::pubcomp_packet properties_don't match its length"
270 );
271 }
272 auto prop_buf = buf.substr(0, property_length_);
273 props_ = make_properties(prop_buf, property_location::pubcomp);
274 buf.remove_prefix(property_length_);
275 }
276 else {
277 throw make_error(
278 errc::bad_message,
279 "v5::pubcomp_packet property_length is invalid"
280 );
281 }
282
283 if (!buf.empty()) {
284 throw make_error(
285 errc::bad_message,
286 "v5::pubcomp_packet properties don't match its length"
287 );
288 }
289}
