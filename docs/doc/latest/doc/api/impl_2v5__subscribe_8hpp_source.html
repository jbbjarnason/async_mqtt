32template <std::size_t PacketIdBytes>
33inline
34basic_subscribe_packet<PacketIdBytes>::basic_subscribe_packet(
35 typename basic_packet_id_type<PacketIdBytes>::type packet_id,
36 std::vector<topic_subopts> params,
37 properties props
38)
39 : fixed_header_{make_fixed_header(control_packet_type::subscribe, 0b0010)},
40 entries_{force_move(params)},
41 remaining_length_{PacketIdBytes},
42 property_length_(async_mqtt::size(props)),
43 props_(force_move(props))
44{
45 using namespace std::literals;
46 topic_length_buf_entries_.reserve(entries_.size());
47 for (auto const& e : entries_) {
48 topic_length_buf_entries_.push_back(
49 endian_static_vector(
50 boost::numeric_cast<std::uint16_t>(e.all_topic().size())
51 )
52 );
53 }
54
55 endian_store(packet_id, packet_id_.data());
56
57 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
58 for (auto e : pb) {
59 property_length_buf_.push_back(e);
60 }
61
62 for (auto const& prop : props_) {
63 auto id = prop.id();
64 if (!validate_property(property_location::subscribe, id)) {
65 throw make_error(
66 errc::bad_message,
67 "v5::subscribe_packet property "s + id_to_str(id) + " is not allowed"
68 );
69 }
70 }
71
72 remaining_length_ += property_length_buf_.size() + property_length_;
73
74 for (auto const& e : entries_) {
75 // reserved bits check
76 if (static_cast<std::uint8_t>(e.opts()) & 0b11000000) {
77 throw make_error(
78 errc::bad_message,
79 "v5::subscribe_packet subopts is invalid"
80 );
81 }
82 switch (e.opts().get_qos()) {
83 case qos::at_most_once:
84 case qos::at_least_once:
85 case qos::exactly_once:
86 break;
87 default:
88 throw make_error(
89 errc::bad_message,
90 "v5::subscribe_packet qos is invalid"
91 );
92 break;
93 }
94 switch (e.opts().get_retain_handling()) {
95 case sub::retain_handling::send:
96 case sub::retain_handling::send_only_new_subscription:
97 case sub::retain_handling::not_send:
98 break;
99 default:
100 throw make_error(
101 errc::bad_message,
102 "v5::subscribe_packet retain_handling is invalid"
103 );
104 break;
105 }
106
107 auto size = e.all_topic().size();
108 if (size > 0xffff) {
109 throw make_error(
110 errc::bad_message,
111 "v5::subscribe_packet length of topic is invalid"
112 );
113 }
114 remaining_length_ +=
115 2 + // topic name length
116 size + // topic filter
117 1; // opts
118
119 if (!utf8string_check(e.all_topic())) {
120 throw make_error(
121 errc::bad_message,
122 "v5::subscribe_packet topic filter invalid utf8"
123 );
124 }
125 }
126
127 remaining_length_buf_ = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
128}

136template <std::size_t PacketIdBytes>
137inline
138std::vector<as::const_buffer> basic_subscribe_packet<PacketIdBytes>::const_buffer_sequence() const {
139 std::vector<as::const_buffer> ret;
140 ret.reserve(num_of_const_buffer_sequence());
141
142 ret.emplace_back(as::buffer(&fixed_header_, 1));
143
144 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
145
146 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
147
148 ret.emplace_back(as::buffer(property_length_buf_.data(), property_length_buf_.size()));
149 auto props_cbs = async_mqtt::const_buffer_sequence(props_);
150 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));
151
152 BOOST_ASSERT(entries_.size() == topic_length_buf_entries_.size());
153 auto it = topic_length_buf_entries_.begin();
154 for (auto const& e : entries_) {
155 ret.emplace_back(as::buffer(it->data(), it->size()));
156 ret.emplace_back(as::buffer(e.all_topic()));
157 ret.emplace_back(as::buffer(&e.opts(), 1));
158 ++it;
159 }
160
161 return ret;
162}

173template <std::size_t PacketIdBytes>
174inline
175std::size_t basic_subscribe_packet<PacketIdBytes>::num_of_const_buffer_sequence() const {
176 return
177 1 + // fixed header
178 1 + // remaining length
179 1 + // packet id
180 [&] () -> std::size_t {
181 if (property_length_buf_.size() == 0) return 0;
182 return
183 1 + // property length
184 async_mqtt::num_of_const_buffer_sequence(props_);
185 }() +
186 entries_.size() * 3; // topic name length, topic name, opts
187}

207template <std::size_t PacketIdBytes>
208inline
209basic_subscribe_packet<PacketIdBytes>::basic_subscribe_packet(buffer buf) {
210 // fixed_header
211 if (buf.empty()) {
212 throw make_error(
213 errc::bad_message,
214 "v5::subscribe_packet fixed_header doesn't exist"
215 );
216 }
217 fixed_header_ = static_cast<std::uint8_t>(buf.front());
218 buf.remove_prefix(1);
219 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
220 if (!cpt_opt || *cpt_opt != control_packet_type::subscribe) {
221 throw make_error(
222 errc::bad_message,
223 "v5::subscribe_packet fixed_header is invalid"
224 );
225 }
226
227 // remaining_length
228 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
229 remaining_length_ = *vl_opt;
230 }
231 else {
232 throw make_error(errc::bad_message, "v5::subscribe_packet remaining length is invalid");
233 }
234 if (remaining_length_ != buf.size()) {
235 throw make_error(errc::bad_message, "v5::subscribe_packet remaining length doesn't match buf.size()");
236 }
237
238 // packet_id
239 if (!copy_advance(buf, packet_id_)) {
240 throw make_error(
241 errc::bad_message,
242 "v5::subscribe_packet packet_id doesn't exist"
243 );
244 }
245
246 // property
247 auto it = buf.begin();
248 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
249 property_length_ = *pl_opt;
250 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
251 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
252 if (buf.size() < property_length_) {
253 throw make_error(
254 errc::bad_message,
255 "v5::subscribe_packet properties_don't match its length"
256 );
257 }
258 auto prop_buf = buf.substr(0, property_length_);
259 props_ = make_properties(prop_buf, property_location::subscribe);
260 buf.remove_prefix(property_length_);
261 }
262 else {
263 throw make_error(
264 errc::bad_message,
265 "v5::subscribe_packet property_length is invalid"
266 );
267 }
268
269 if (remaining_length_ == 0) {
270 throw make_error(errc::bad_message, "v5::subscribe_packet doesn't have entries");
271 }
272
273 while (!buf.empty()) {
274 // topic_length
275 static_vector<char, 2> topic_length_buf;
276 if (!insert_advance(buf, topic_length_buf)) {
277 throw make_error(
278 errc::bad_message,
279 "v5::subscribe_packet length of topic is invalid"
280 );
281 }
282 auto topic_length = endian_load<std::uint16_t>(topic_length_buf.data());
283 topic_length_buf_entries_.push_back(topic_length_buf);
284
285 // topic
286 if (buf.size() < topic_length) {
287 throw make_error(
288 errc::bad_message,
289 "v5::subscribe_packet topic doesn't match its length"
290 );
291 }
292 auto topic = buf.substr(0, topic_length);
293
294 if (!utf8string_check(topic)) {
295 throw make_error(
296 errc::bad_message,
297 "v5::subscribe_packet topic filter invalid utf8"
298 );
299 }
300
301 buf.remove_prefix(topic_length);
302
303 // opts
304 if (buf.empty()) {
305 throw make_error(
306 errc::bad_message,
307 "v5::subscribe_packet subscribe options doesn't exist"
308 );
309 }
310 auto opts = static_cast<sub::opts>(std::uint8_t(buf.front()));
311 // reserved bits check
312 if (static_cast<std::uint8_t>(opts) & 0b11000000) {
313 throw make_error(
314 errc::bad_message,
315 "v5::subscribe_packet subopts is invalid"
316 );
317 }
318 switch (opts.get_qos()) {
319 case qos::at_most_once:
320 case qos::at_least_once:
321 case qos::exactly_once:
322 break;
323 default:
324 throw make_error(
325 errc::bad_message,
326 "v5::subscribe_packet qos is invalid"
327 );
328 break;
329 }
330 switch (opts.get_retain_handling()) {
331 case sub::retain_handling::send:
332 case sub::retain_handling::send_only_new_subscription:
333 case sub::retain_handling::not_send:
334 break;
335 default:
336 throw make_error(
337 errc::bad_message,
338 "v5::subscribe_packet retain_handling is invalid"
339 );
340 break;
341 }
342 entries_.emplace_back(std::string{topic}, opts);
343 buf.remove_prefix(1);
344 }
345}

350 o <<
351 "v5::subscribe{" <<
352 "pid:" << v.packet_id() << ",[";

355 if (b != e) {
356 o <<
357 "{"
358 "topic:" << b->topic() << "," <<
359 "sn:" << b->sharename() << "," <<
360 "qos:" << b->opts().get_qos() << "," <<
361 "rh:" << b->opts().get_retain_handling() << "," <<
362 "nl:" << b->opts().get_nl() << "," <<
363 "rap:" << b->opts().get_rap() <<
364 "}";
365 ++b;
366 }
367 for (; b != e; ++b) {
368 o << "," <<
369 "{"
370 "topic:" << b->topic() << "," <<
371 "sn:" << b->sharename() << "," <<
372 "qos:" << b->opts().get_qos() << "," <<
373 "rh:" << b->opts().get_retain_handling() << "," <<
374 "nl:" << b->opts().get_nl() << "," <<
375 "rap:" << b->opts().get_rap() <<
376 "}";
377 }
378 o << "]";
379 if (!v.props().empty()) {
380 o << ",ps:" << v.props();
381 };
382 o << "}";
383 return o;
384}
