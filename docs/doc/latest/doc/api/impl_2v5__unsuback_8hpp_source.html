29template <std::size_t PacketIdBytes>
30inline
31basic_unsuback_packet<PacketIdBytes>::basic_unsuback_packet(
32 typename basic_packet_id_type<PacketIdBytes>::type packet_id,
33 std::vector<unsuback_reason_code> params,
34 properties props
35)
36 : fixed_header_{make_fixed_header(control_packet_type::unsuback, 0b0000)},
37 entries_{force_move(params)},
38 remaining_length_{PacketIdBytes + entries_.size()},
39 property_length_(async_mqtt::size(props)),
40 props_(force_move(props))
41{
42 using namespace std::literals;
43 endian_store(packet_id, packet_id_.data());
44
45 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
46 for (auto e : pb) {
47 property_length_buf_.push_back(e);
48 }
49
50 for (auto const& prop : props_) {
51 auto id = prop.id();
52 if (!validate_property(property_location::unsuback, id)) {
53 throw make_error(
54 errc::bad_message,
55 "v5::unsuback_packet property "s + id_to_str(id) + " is not allowed"
56 );
57 }
58 }
59
60 remaining_length_ += property_length_buf_.size() + property_length_;
61 remaining_length_buf_ = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
62}

70template <std::size_t PacketIdBytes>
71inline
72std::vector<as::const_buffer> basic_unsuback_packet<PacketIdBytes>::const_buffer_sequence() const {
73 std::vector<as::const_buffer> ret;
74 ret.reserve(num_of_const_buffer_sequence());
75
76 ret.emplace_back(as::buffer(&fixed_header_, 1));
77
78 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
79
80 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
81
82 ret.emplace_back(as::buffer(property_length_buf_.data(), property_length_buf_.size()));
83 auto props_cbs = async_mqtt::const_buffer_sequence(props_);
84 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));
85
86 ret.emplace_back(as::buffer(entries_.data(), entries_.size()));
87
88 return ret;
89}

100template <std::size_t PacketIdBytes>
101inline
102std::size_t basic_unsuback_packet<PacketIdBytes>::num_of_const_buffer_sequence() const {
103 return
104 1 + // fixed header
105 1 + // remaining length
106 1 + // packet id
107 [&] () -> std::size_t {
108 if (property_length_buf_.size() == 0) return 0;
109 return
110 1 + // property length
111 async_mqtt::num_of_const_buffer_sequence(props_);
112 }() +
113 1; // unsuback_reason_code vector
114}

134template <std::size_t PacketIdBytes>
135inline
136basic_unsuback_packet<PacketIdBytes>::basic_unsuback_packet(buffer buf) {
137 // fixed_header
138 if (buf.empty()) {
139 throw make_error(
140 errc::bad_message,
141 "v5::unsuback_packet fixed_header doesn't exist"
142 );
143 }
144 fixed_header_ = static_cast<std::uint8_t>(buf.front());
145 buf.remove_prefix(1);
146 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
147 if (!cpt_opt || *cpt_opt != control_packet_type::unsuback) {
148 throw make_error(
149 errc::bad_message,
150 "v5::unsuback_packet fixed_header is invalid"
151 );
152 }
153
154 // remaining_length
155 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
156 remaining_length_ = *vl_opt;
157 }
158 else {
159 throw make_error(errc::bad_message, "v5::unsuback_packet remaining length is invalid");
160 }
161 if (remaining_length_ != buf.size()) {
162 throw make_error(errc::bad_message, "v5::unsuback_packet remaining length doesn't match buf.size()");
163 }
164
165 // packet_id
166 if (!copy_advance(buf, packet_id_)) {
167 throw make_error(
168 errc::bad_message,
169 "v5::unsuback_packet packet_id doesn't exist"
170 );
171 }
172
173 // property
174 auto it = buf.begin();
175 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
176 property_length_ = *pl_opt;
177 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
178 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
179 if (buf.size() < property_length_) {
180 throw make_error(
181 errc::bad_message,
182 "v5::unsuback_packet properties_don't match its length"
183 );
184 }
185 auto prop_buf = buf.substr(0, property_length_);
186 props_ = make_properties(prop_buf, property_location::unsuback);
187 buf.remove_prefix(property_length_);
188 }
189 else {
190 throw make_error(
191 errc::bad_message,
192 "v5::unsuback_packet property_length is invalid"
193 );
194 }
195
196 if (remaining_length_ == 0) {
197 throw make_error(errc::bad_message, "v5::unsuback_packet doesn't have entries");
198 }
199
200 while (!buf.empty()) {
201 // reason_code
202 if (buf.empty()) {
203 throw make_error(
204 errc::bad_message,
205 "v5::unsuback_packet unsuback_reason_code doesn't exist"
206 );
207 }
208 auto rc = static_cast<unsuback_reason_code>(buf.front());
209 entries_.emplace_back(rc);
210 buf.remove_prefix(1);
211 }
212}
