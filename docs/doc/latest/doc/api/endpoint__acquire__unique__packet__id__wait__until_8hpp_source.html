14template <role Role, std::size_t PacketIdBytes, typename NextLayer>
15struct basic_endpoint<Role, PacketIdBytes, NextLayer>::
16acquire_unique_packet_id_wait_until_op {
17 this_type& ep;
18 this_type_wp retry_wp = ep.weak_from_this();
19 std::optional<typename basic_packet_id_type<PacketIdBytes>::type> pid_opt = std::nullopt;
20 enum { dispatch, complete } state = dispatch;
21
22 template <typename Self>
23 void operator()(
24 Self& self,
25 error_code ec = error_code{}
26 ) {
27 if (retry_wp.expired()) return;
28 switch (state) {
29 case dispatch: {
30 state = complete;
31 auto& a_ep{ep};
32 as::dispatch(
33 as::bind_executor(
34 a_ep.get_executor(),
35 force_move(self)
36 )
37 );
38 } break;
39 case complete: {
40 auto acq_proc =
41 [&] {
42 pid_opt = ep.pid_man_.acquire_unique_id();
43 if (pid_opt) {
44 self.complete(*pid_opt);
45 }
46 else {
47 ASYNC_MQTT_LOG("mqtt_impl", warning)
48 << ASYNC_MQTT_ADD_VALUE(address, &ep)
49 << "packet_id is fully allocated. waiting release";
50 // infinity timer. cancel is retry trigger.
51 auto& a_ep{ep};
52 a_ep.add_retry(
53 as::bind_executor(
54 a_ep.get_executor(),
55 force_move(self)
56 )
57 );
58 }
59 };
60
61 if (ec == errc::operation_canceled) {
62 ep.complete_retry_one();
63 acq_proc();
64 }
65 else if (ep.has_retry()) {
66 ASYNC_MQTT_LOG("mqtt_impl", warning)
67 << ASYNC_MQTT_ADD_VALUE(address, &ep)
68 << "packet_id waiter exists. add the end of waiter queue";
69 // infinity timer. cancel is retry trigger.
70 auto& a_ep{ep};
71 a_ep.add_retry(
72 as::bind_executor(
73 a_ep.get_executor(),
74 force_move(self)
75 )
76 );
77 }
78 else {
79 acq_proc();
80 }
81 } break;
82 }
83 }
84};
