75template <std::size_t PacketIdBytes>
76inline
77basic_puback_packet<PacketIdBytes>::basic_puback_packet(buffer buf) {
78 // fixed_header
79 if (buf.empty()) {
80 throw make_error(
81 errc::bad_message,
82 "v3_1_1::puback_packet fixed_header doesn't exist"
83 );
84 }
85 all_.push_back(buf.front());
86 buf.remove_prefix(1);
87 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(all_.back()));
88 if (!cpt_opt || *cpt_opt != control_packet_type::puback) {
89 throw make_error(
90 errc::bad_message,
91 "v3_1_1::puback_packet fixed_header is invalid"
92 );
93 }
94
95 // remaining_length
96 if (buf.empty()) {
97 throw make_error(
98 errc::bad_message,
99 "v3_1_1::puback_packet remaining_length doesn't exist"
100 );
101 }
102 all_.push_back(buf.front());
103 buf.remove_prefix(1);
104 if (static_cast<std::uint8_t>(all_.back()) != PacketIdBytes) {
105 throw make_error(
106 errc::bad_message,
107 "v3_1_1::puback_packet remaining_length is invalid"
108 );
109 }
110
111 // variable header
112 if (buf.size() != PacketIdBytes) {
113 throw make_error(
114 errc::bad_message,
115 "v3_1_1::puback_packet variable header doesn't match its length"
116 );
117 }
118 std::copy(buf.begin(), buf.end(), std::back_inserter(all_));
119}
