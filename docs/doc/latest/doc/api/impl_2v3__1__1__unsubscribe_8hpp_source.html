31template <std::size_t PacketIdBytes>
32inline
33basic_unsubscribe_packet<PacketIdBytes>::basic_unsubscribe_packet(
34 typename basic_packet_id_type<PacketIdBytes>::type packet_id,
35 std::vector<topic_sharename> params
36)
37 : fixed_header_{make_fixed_header(control_packet_type::unsubscribe, 0b0010)},
38 entries_{force_move(params)},
39 remaining_length_{PacketIdBytes}
40{
41 topic_length_buf_entries_.reserve(entries_.size());
42 for (auto const& e : entries_) {
43 topic_length_buf_entries_.push_back(
44 endian_static_vector(
45 boost::numeric_cast<std::uint16_t>(e.all_topic().size())
46 )
47 );
48 }
49
50 endian_store(packet_id, packet_id_.data());
51
52 // Check for errors before allocating.
53 for (auto const& e : entries_) {
54 auto size = e.all_topic().size();
55 if (size > 0xffff) {
56 throw make_error(
57 errc::bad_message,
58 "v3_1_1::unsubscribe_packet length of topic is invalid"
59 );
60 }
61 remaining_length_ +=
62 2 + // topic filter length
63 size; // topic filter
64
65 if (!utf8string_check(e.all_topic())) {
66 throw make_error(
67 errc::bad_message,
68 "v3_1_1::unsubscribe_packet topic filter invalid utf8"
69 );
70 }
71 }
72
73 remaining_length_buf_ = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
74}

82template <std::size_t PacketIdBytes>
83inline
84std::vector<as::const_buffer> basic_unsubscribe_packet<PacketIdBytes>::const_buffer_sequence() const {
85 std::vector<as::const_buffer> ret;
86 ret.reserve(num_of_const_buffer_sequence());
87
88 ret.emplace_back(as::buffer(&fixed_header_, 1));
89
90 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
91
92 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
93
94 BOOST_ASSERT(entries_.size() == topic_length_buf_entries_.size());
95 auto it = topic_length_buf_entries_.begin();
96 for (auto const& e : entries_) {
97 ret.emplace_back(as::buffer(it->data(), it->size()));
98 ret.emplace_back(as::buffer(e.all_topic()));
99 ++it;
100 }
101
102 return ret;
103}

136template <std::size_t PacketIdBytes>
137inline
138basic_unsubscribe_packet<PacketIdBytes>::basic_unsubscribe_packet(buffer buf) {
139 // fixed_header
140 if (buf.empty()) {
141 throw make_error(
142 errc::bad_message,
143 "v3_1_1::unsubscribe_packet fixed_header doesn't exist"
144 );
145 }
146 fixed_header_ = static_cast<std::uint8_t>(buf.front());
147 buf.remove_prefix(1);
148 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
149 if (!cpt_opt || *cpt_opt != control_packet_type::unsubscribe) {
150 throw make_error(
151 errc::bad_message,
152 "v3_1_1::unsubscribe_packet fixed_header is invalid"
153 );
154 }
155
156 // remaining_length
157 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
158 remaining_length_ = *vl_opt;
159 }
160 else {
161 throw make_error(errc::bad_message, "v3_1_1::unsubscribe_packet remaining length is invalid");
162 }
163 if (remaining_length_ != buf.size()) {
164 throw make_error(errc::bad_message, "v3_1_1::unsubscribe_packet remaining length doesn't match buf.size()");
165 }
166
167 // packet_id
168 if (!copy_advance(buf, packet_id_)) {
169 throw make_error(
170 errc::bad_message,
171 "v3_1_1::unsubscribe_packet packet_id doesn't exist"
172 );
173 }
174
175 if (remaining_length_ == 0) {
176 throw make_error(errc::bad_message, "v3_1_1::unsubscribe_packet doesn't have entries");
177 }
178
179 while (!buf.empty()) {
180 // topic_length
181 static_vector<char, 2> topic_length_buf;
182 if (!insert_advance(buf, topic_length_buf)) {
183 throw make_error(
184 errc::bad_message,
185 "v3_1_1::unsubscribe_packet length of topic is invalid"
186 );
187 }
188 auto topic_length = endian_load<std::uint16_t>(topic_length_buf.data());
189 topic_length_buf_entries_.push_back(topic_length_buf);
190
191 // topic
192 if (buf.size() < topic_length) {
193 throw make_error(
194 errc::bad_message,
195 "v3_1_1::unsubscribe_packet topic doesn't match its length"
196 );
197 }
198 auto topic = buf.substr(0, topic_length);
199 if (!utf8string_check(topic)) {
200 throw make_error(
201 errc::bad_message,
202 "v3_1_1::unsubscribe_packet topic filter invalid utf8"
203 );
204 }
205 entries_.emplace_back(std::string{topic});
206 buf.remove_prefix(topic_length);
207 }
208}
