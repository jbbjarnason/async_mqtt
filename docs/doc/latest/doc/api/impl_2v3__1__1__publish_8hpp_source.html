40template <std::size_t PacketIdBytes>
41template <
42 typename StringViewLike,
43 typename BufferSequence,
44 std::enable_if_t<
45 std::is_convertible_v<std::decay_t<StringViewLike>, std::string_view> &&
46 (
47 is_buffer_sequence<std::decay_t<BufferSequence>>::value ||
48 std::is_convertible_v<std::decay_t<BufferSequence>, std::string_view>
49 ),
50 std::nullptr_t
51 >
52>
53inline
54basic_publish_packet<PacketIdBytes>::basic_publish_packet(
55 typename basic_packet_id_type<PacketIdBytes>::type packet_id,
56 StringViewLike&& topic_name,
57 BufferSequence&& payloads,
58 pub::opts pubopts
59)
60 : fixed_header_(
61 make_fixed_header(control_packet_type::publish, 0b0000) | std::uint8_t(pubopts)
62 ),
63 topic_name_{std::string{std::forward<StringViewLike>(topic_name)}},
64 packet_id_(PacketIdBytes),
65 remaining_length_(
66 2 // topic name length
67 + topic_name_.size() // topic name
68 + ( (pubopts.get_qos() == qos::at_least_once || pubopts.get_qos() == qos::exactly_once)
69 ? PacketIdBytes // packet_id
70 : 0)
71 )
72{
73 topic_name_length_buf_.resize(topic_name_length_buf_.capacity());
74 endian_store(
75 boost::numeric_cast<std::uint16_t>(topic_name_.size()),
76 topic_name_length_buf_.data()
77 );
78
79 if constexpr (std::is_convertible_v<std::decay_t<BufferSequence>, std::string_view>) {
80 remaining_length_ += std::string_view(payloads).size();
81 payloads_.emplace_back(buffer{std::string{payloads}});
82 }
83 else {
84 auto b = buffer_sequence_begin(payloads);
85 auto e = buffer_sequence_end(payloads);
86 auto num_of_payloads = static_cast<std::size_t>(std::distance(b, e));
87 payloads_.reserve(num_of_payloads);
88 for (; b != e; ++b) {
89 auto const& payload = *b;
90 remaining_length_ += payload.size();
91 payloads_.push_back(payload);
92 }
93 }
94
95 if (!utf8string_check(topic_name_)) {
96 throw make_error(
97 errc::bad_message,
98 "v3_1_1::publish_packet topic name invalid utf8"
99 );
100 }
101
102 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
103 for (auto e : rb) {
104 remaining_length_buf_.push_back(e);
105 }
106 switch (pubopts.get_qos()) {
107 case qos::at_most_once:
108 if (packet_id != 0) {
109 throw make_error(
110 errc::bad_message,
111 "v3_1_1::publish_packet qos0 but non 0 packet_id"
112 );
113 }
114 endian_store(0, packet_id_.data());
115 break;
116 case qos::at_least_once:
117 case qos::exactly_once:
118 if (packet_id == 0) {
119 throw make_error(
120 errc::bad_message,
121 "v3_1_1::publish_packet qos not 0 but packet_id is 0"
122 );
123 }
124 endian_store(packet_id, packet_id_.data());
125 break;
126 default:
127 throw make_error(
128 errc::bad_message,
129 "v3_1_1::publish_packet qos is invalid"
130 );
131 break;
132 }
133}

168template <std::size_t PacketIdBytes>
169inline
170std::vector<as::const_buffer> basic_publish_packet<PacketIdBytes>::const_buffer_sequence() const {
171 std::vector<as::const_buffer> ret;
172 ret.reserve(num_of_const_buffer_sequence());
173 ret.emplace_back(as::buffer(&fixed_header_, 1));
174 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
175 ret.emplace_back(as::buffer(topic_name_length_buf_.data(), topic_name_length_buf_.size()));
176 ret.emplace_back(as::buffer(topic_name_));
177 if (packet_id() != 0) {
178 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
179 }
180 for (auto const& payload : payloads_) {
181 ret.emplace_back(as::buffer(payload));
182 }
183 return ret;
184}

258template <std::size_t PacketIdBytes>
259inline
260basic_publish_packet<PacketIdBytes>::basic_publish_packet(buffer buf)
261 : packet_id_(PacketIdBytes) {
262 // fixed_header
263 if (buf.empty()) {
264 throw make_error(
265 errc::bad_message,
266 "v3_1_1::publish_packet fixed_header doesn't exist"
267 );
268 }
269 fixed_header_ = static_cast<std::uint8_t>(buf.front());
270 auto qos_value = pub::get_qos(fixed_header_);
271 buf.remove_prefix(1);
272 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
273 if (!cpt_opt || *cpt_opt != control_packet_type::publish) {
274 throw make_error(
275 errc::bad_message,
276 "v3_1_1::publish_packet fixed_header is invalid"
277 );
278 }
279
280 // remaining_length
281 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
282 remaining_length_ = *vl_opt;
283 }
284 else {
285 throw make_error(errc::bad_message, "v3_1_1::publish_packet remaining length is invalid");
286 }
287 if (remaining_length_ != buf.size()) {
288 throw make_error(errc::bad_message, "v3_1_1::publish_packet remaining length doesn't match buf.size()");
289 }
290
291 // topic_name_length
292 if (!insert_advance(buf, topic_name_length_buf_)) {
293 throw make_error(
294 errc::bad_message,
295 "v3_1_1::publish_packet length of topic_name is invalid"
296 );
297 }
298 auto topic_name_length = endian_load<std::uint16_t>(topic_name_length_buf_.data());
299
300 // topic_name
301 if (buf.size() < topic_name_length) {
302 throw make_error(
303 errc::bad_message,
304 "v3_1_1::publish_packet topic_name doesn't match its length"
305 );
306 }
307 topic_name_ = buf.substr(0, topic_name_length);
308
309 if (!utf8string_check(topic_name_)) {
310 throw make_error(
311 errc::bad_message,
312 "v3_1_1::publish_packet topic name invalid utf8"
313 );
314 }
315
316 buf.remove_prefix(topic_name_length);
317
318 // packet_id
319 switch (qos_value) {
320 case qos::at_most_once:
321 endian_store(typename basic_packet_id_type<PacketIdBytes>::type{0}, packet_id_.data());
322 break;
323 case qos::at_least_once:
324 case qos::exactly_once:
325 if (!copy_advance(buf, packet_id_)) {
326 throw make_error(
327 errc::bad_message,
328 "v3_1_1::publish_packet packet_id doesn't exist"
329 );
330 }
331 break;
332 default:
333 throw make_error(
334 errc::bad_message,
335 "v3_1_1::publish_packet qos is invalid"
336 );
337 break;
338 };
339
340 // payload
341 if (!buf.empty()) {
342 payloads_.emplace_back(force_move(buf));
343 }
344}
