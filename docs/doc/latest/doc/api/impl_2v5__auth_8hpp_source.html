71std::vector<as::const_buffer> auth_packet::const_buffer_sequence() const {
72 std::vector<as::const_buffer> ret;
73 ret.reserve(num_of_const_buffer_sequence());
74 ret.emplace_back(as::buffer(&fixed_header_, 1));
75 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
76
77 if (reason_code_) {
78 ret.emplace_back(as::buffer(&*reason_code_, 1));
79
80 if (property_length_buf_.size() != 0) {
81 ret.emplace_back(as::buffer(property_length_buf_.data(), property_length_buf_.size()));
82 auto props_cbs = async_mqtt::const_buffer_sequence(props_);
83 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));
84 }
85 }
86
87 return ret;
88}

99std::size_t auth_packet::num_of_const_buffer_sequence() const {
100 return
101 1 + // fixed header
102 1 + // remaining length
103 [&] () -> std::size_t {
104 if (reason_code_) {

106 1 + // reason_code

108 if (property_length_buf_.size() == 0) return 0;

129inline
130auth_packet::auth_packet(
131 std::optional<auth_reason_code> reason_code,
132 properties props
133)
134 : fixed_header_{
135 make_fixed_header(control_packet_type::auth, 0b0000)
136 },
137 remaining_length_{
138 0
139 },
140 reason_code_{reason_code},
141 property_length_(async_mqtt::size(props)),
142 props_(force_move(props))
143{
144 using namespace std::literals;
145
146 auto guard = unique_scope_guard(
147 [&] {
148 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
149 for (auto e : rb) {
150 remaining_length_buf_.push_back(e);
151 }
152 }
153 );
154
155 if (!reason_code_) return;
156 remaining_length_ += 1;
157
158 if (property_length_ == 0) return;
159
160 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
161 for (auto e : pb) {
162 property_length_buf_.push_back(e);
163 }
164
165 for (auto const& prop : props_) {
166 auto id = prop.id();
167 if (!validate_property(property_location::auth, id)) {
168 throw make_error(
169 errc::bad_message,
170 "v5::auth_packet property "s + id_to_str(id) + " is not allowed"
171 );
172 }
173 }
174
175 remaining_length_ += property_length_buf_.size() + property_length_;
176}

178inline
179auth_packet::auth_packet(buffer buf) {
180 // fixed_header
181 if (buf.empty()) {
182 throw make_error(
183 errc::bad_message,
184 "v5::auth_packet fixed_header doesn't exist"
185 );
186 }
187 fixed_header_ = static_cast<std::uint8_t>(buf.front());
188 buf.remove_prefix(1);
189
190 // remaining_length
191 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
192 remaining_length_ = *vl_opt;
193 }
194 else {
195 throw make_error(errc::bad_message, "v5::auth_packet remaining length is invalid");
196 }
197
198 if (remaining_length_ == 0) {
199 if (!buf.empty()) {
200 throw make_error(errc::bad_message, "v5::auth_packet remaining length is invalid");
201 }
202 return;
203 }
204
205 // connect_reason_code
206 reason_code_.emplace(static_cast<auth_reason_code>(buf.front()));
207 buf.remove_prefix(1);
208 switch (*reason_code_) {
209 case auth_reason_code::success:
210 case auth_reason_code::continue_authentication:
211 case auth_reason_code::re_authenticate:
212 break;
213 default:
214 throw make_error(
215 errc::bad_message,
216 "v5::auth_packet connect reason_code is invalid"
217 );
218 break;
219 }
220
221 if (remaining_length_ == 1) {
222 if (!buf.empty()) {
223 throw make_error(errc::bad_message, "v5::auth_packet remaining length is invalid");
224 }
225 return;
226 }
227
228 // property
229 auto it = buf.begin();
230 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
231 property_length_ = *pl_opt;
232 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
233 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
234 if (buf.size() < property_length_) {
235 throw make_error(
236 errc::bad_message,
237 "v5::auth_packet properties_don't match its length"
238 );
239 }
240 auto prop_buf = buf.substr(0, property_length_);
241 props_ = make_properties(prop_buf, property_location::auth);
242 buf.remove_prefix(property_length_);
243 }
244 else {
245 throw make_error(
246 errc::bad_message,
247 "v5::auth_packet property_length is invalid"
248 );
249 }
250
251 if (!buf.empty()) {
252 throw make_error(
253 errc::bad_message,
254 "v5::auth_packet properties don't match its length"
255 );
256 }
257}
