60inline
61pingresp_packet::pingresp_packet(buffer buf) {
62 // fixed_header
63 if (buf.empty()) {
64 throw make_error(
65 errc::bad_message,
66 "v3_1_1::pingresp_packet fixed_header doesn't exist"
67 );
68 }
69 all_.push_back(buf.front());
70 buf.remove_prefix(1);
71 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(all_.back()));
72 if (!cpt_opt || *cpt_opt != control_packet_type::pingresp) {
73 throw make_error(
74 errc::bad_message,
75 "v3_1_1::pingresp_packet fixed_header is invalid"
76 );
77 }
78
79 // remaining_length
80 if (buf.empty()) {
81 throw make_error(
82 errc::bad_message,
83 "v3_1_1::pingresp_packet remaining_length doesn't exist"
84 );
85 }
86 all_.push_back(buf.front());
87
88 if (static_cast<std::uint8_t>(all_.back()) != 0) {
89 throw make_error(
90 errc::bad_message,
91 "v3_1_1::pingresp_packet remaining_length is invalid"
92 );
93 }
94}
