83template <std::size_t N>
84struct n_bytes_property : private boost::totally_ordered<n_bytes_property<N>> {
85 n_bytes_property(property::id id, static_vector<char, N> const& buf)
86 : id_{id},
87 buf_{buf} // very small size copy
88 {
89 }
90
91 template <typename It, typename End>
92 n_bytes_property(property::id id, It b, End e)
93 :id_{id}, buf_(b, e) {}
94
95 n_bytes_property(property::id id, buffer const& buf)
96 :id_{id} {
97 BOOST_ASSERT(buf.size() >= N);
98 buf_.insert(buf_.end(), (buf.begin(), std::next(buf.begin(), N)));
99 }
100
105 std::vector<as::const_buffer> const_buffer_sequence() const {
106 std::vector<as::const_buffer> v;
107 v.reserve(num_of_const_buffer_sequence());
108 v.emplace_back(as::buffer(&id_, 1));
109 v.emplace_back(as::buffer(buf_.data(), buf_.size()));
110 return v;
111 }
112
117 property::id id() const {
118 return id_;
119 }
120
125 std::size_t size() const {
126 return 1 + buf_.size();
127 }
128
133 static constexpr std::size_t num_of_const_buffer_sequence() {
134 return 2;
135 }
136
137 friend bool operator<(n_bytes_property<N> const& lhs, n_bytes_property<N> const& rhs) {
138 return std::tie(lhs.id_, lhs.buf_) < std::tie(rhs.id_, rhs.buf_);
139 }
140
141 friend bool operator==(n_bytes_property<N> const& lhs, n_bytes_property<N> const& rhs) {
142 return std::tie(lhs.id_, lhs.buf_) == std::tie(rhs.id_, rhs.buf_);
143 }
144
145 static constexpr ostream_format const of_ = ostream_format::direct;
146 property::id id_;
147 static_vector<char, N> buf_;
148};

154struct binary_property : private boost::totally_ordered<binary_property> {
155 binary_property(property::id id, buffer buf)
156 :id_{id},
157 buf_{force_move(buf)},
158 length_(2) // size 2
159 {
160 if (buf_.size() > 0xffff) {
161 throw make_error(
162 errc::bad_message,
163 "property::binary_property length is invalid"
164 );
165 }
166 endian_store(boost::numeric_cast<std::uint16_t>(buf_.size()), length_.data());
167 }
168
173 std::vector<as::const_buffer> const_buffer_sequence() const {
174 std::vector<as::const_buffer> v;
175 v.reserve(num_of_const_buffer_sequence());
176 v.emplace_back(as::buffer(&id_, 1));
177 v.emplace_back(as::buffer(length_.data(), length_.size()));
178 v.emplace_back(as::buffer(buf_.data(), buf_.size()));
179 return v;
180 }
181
186 property::id id() const {
187 return id_;
188 }
189
194 std::size_t size() const {
195 return 1 + length_.size() + buf_.size();
196 }
197
202 static constexpr std::size_t num_of_const_buffer_sequence() {
203 return 3;
204 }
205
210 constexpr buffer const& val() const {
211 return buf_;
212 }
213
214 friend bool operator<(binary_property const& lhs, binary_property const& rhs) {
215 return std::tie(lhs.id_, lhs.buf_) < std::tie(rhs.id_, rhs.buf_);
216 }
217
218 friend bool operator==(binary_property const& lhs, binary_property const& rhs) {
219 return std::tie(lhs.id_, lhs.buf_) == std::tie(rhs.id_, rhs.buf_);
220 }
221
222 static constexpr ostream_format const of_ = ostream_format::json_like;
223 property::id id_;
224 buffer buf_;
225 static_vector<char, 2> length_;
226};

267
272 property::id id() const {
273 return id_;

275
280 std::size_t size() const {
281 return 1 + value_.size();
282 }
283
288 static constexpr std::size_t num_of_const_buffer_sequence() {
289 return 2;
290 }
291
296 std::size_t val() const {
297 auto it{value_.begin()};
298 auto val_opt{variable_bytes_to_val(it, value_.end())};
299 BOOST_ASSERT(val_opt);

301 }
302

316 property::id id_;
