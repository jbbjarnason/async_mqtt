127template <std::size_t N>
128struct n_bytes_property : private boost::totally_ordered<n_bytes_property<N>> {
129 n_bytes_property(property::id id, static_vector<char, N> const& buf)
130 : id_{id},
131 buf_{buf} // very small size copy
132 {
133 }
134
135 template <typename It, typename End>
136 n_bytes_property(property::id id, It b, End e)
137 :id_{id}, buf_(b, e) {}
138
139 n_bytes_property(property::id id, buffer const& buf)
140 :id_{id} {
141 BOOST_ASSERT(buf.size() >= N);
142 buf_.insert(buf_.end(), (buf.begin(), std::next(buf.begin(), N)));
143 }
144
149 std::vector<as::const_buffer> const_buffer_sequence() const {
150 std::vector<as::const_buffer> v;
151 v.reserve(num_of_const_buffer_sequence());
152 v.emplace_back(as::buffer(&id_, 1));
153 v.emplace_back(as::buffer(buf_.data(), buf_.size()));
154 return v;
155 }
156
161 property::id id() const {
162 return id_;
163 }
164
169 std::size_t size() const {
170 return 1 + buf_.size();
171 }
172
177 static constexpr std::size_t num_of_const_buffer_sequence() {
178 return 2;
179 }
180
181 friend bool operator<(n_bytes_property<N> const& lhs, n_bytes_property<N> const& rhs) {
182 return std::tie(lhs.id_, lhs.buf_) < std::tie(rhs.id_, rhs.buf_);
183 }
184
185 friend bool operator==(n_bytes_property<N> const& lhs, n_bytes_property<N> const& rhs) {
186 return std::tie(lhs.id_, lhs.buf_) == std::tie(rhs.id_, rhs.buf_);
187 }
188
189 static constexpr ostream_format const of_ = ostream_format::direct;
190 property::id id_;
191 static_vector<char, N> buf_;
192};

198struct binary_property : private boost::totally_ordered<binary_property> {
199 binary_property(property::id id, buffer buf)
200 :id_{id},
201 buf_{force_move(buf)},
202 length_(2) // size 2
203 {
204 if (buf_.size() > 0xffff) {
205 throw make_error(
206 errc::bad_message,
207 "property::binary_property length is invalid"
208 );
209 }
210 endian_store(boost::numeric_cast<std::uint16_t>(buf_.size()), length_.data());
211 }
212
217 std::vector<as::const_buffer> const_buffer_sequence() const {
218 std::vector<as::const_buffer> v;
219 v.reserve(num_of_const_buffer_sequence());
220 v.emplace_back(as::buffer(&id_, 1));
221 v.emplace_back(as::buffer(length_.data(), length_.size()));
222 v.emplace_back(as::buffer(buf_.data(), buf_.size()));
223 return v;
224 }
225
230 property::id id() const {
231 return id_;
232 }
233
238 std::size_t size() const {
239 return 1 + length_.size() + buf_.size();
240 }
241
246 static constexpr std::size_t num_of_const_buffer_sequence() {
247 return 3;
248 }
249
254 constexpr buffer const& val() const {
255 return buf_;
256 }
257
258 friend bool operator<(binary_property const& lhs, binary_property const& rhs) {
259 return std::tie(lhs.id_, lhs.buf_) < std::tie(rhs.id_, rhs.buf_);
260 }
261
262 friend bool operator==(binary_property const& lhs, binary_property const& rhs) {
263 return std::tie(lhs.id_, lhs.buf_) == std::tie(rhs.id_, rhs.buf_);
264 }
265
266 static constexpr ostream_format const of_ = ostream_format::json_like;
267 property::id id_;
268 buffer buf_;
269 static_vector<char, 2> length_;
270};
