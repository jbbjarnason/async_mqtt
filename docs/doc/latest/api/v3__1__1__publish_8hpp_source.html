85 template <
86 typename StringViewLike,
87 typename BufferSequence,
88 std::enable_if_t<
89 std::is_convertible_v<std::decay_t<StringViewLike>, std::string_view> &&
90 (
91 is_buffer_sequence<std::decay_t<BufferSequence>>::value ||
92 std::is_convertible_v<std::decay_t<BufferSequence>, std::string_view>
93 ),
94 std::nullptr_t
95 > = nullptr
96 >
97 basic_publish_packet(
98 packet_id_t packet_id,
99 StringViewLike&& topic_name,
100 BufferSequence&& payloads,
101 pub::opts pubopts
102 )
103 : fixed_header_(
104 make_fixed_header(control_packet_type::publish, 0b0000) | std::uint8_t(pubopts)
105 ),
106 topic_name_{std::string{std::forward<StringViewLike>(topic_name)}},
107 packet_id_(PacketIdBytes),
108 remaining_length_(
109 2 // topic name length
110 + topic_name_.size() // topic name
111 + ( (pubopts.get_qos() == qos::at_least_once || pubopts.get_qos() == qos::exactly_once)
112 ? PacketIdBytes // packet_id
113 : 0)
114 )
115 {
116 topic_name_length_buf_.resize(topic_name_length_buf_.capacity());
117 endian_store(
118 boost::numeric_cast<std::uint16_t>(topic_name_.size()),
119 topic_name_length_buf_.data()
120 );
121
122 if constexpr (std::is_convertible_v<std::decay_t<BufferSequence>, std::string_view>) {
123 remaining_length_ += std::string_view(payloads).size();
124 payloads_.emplace_back(buffer{std::string{payloads}});
125 }
126 else {
127 auto b = buffer_sequence_begin(payloads);
128 auto e = buffer_sequence_end(payloads);
129 auto num_of_payloads = static_cast<std::size_t>(std::distance(b, e));
130 payloads_.reserve(num_of_payloads);
131 for (; b != e; ++b) {
132 auto const& payload = *b;
133 remaining_length_ += payload.size();
134 payloads_.push_back(payload);
135 }
136 }
137
138 if (!utf8string_check(topic_name_)) {
139 throw make_error(
140 errc::bad_message,
141 "v3_1_1::publish_packet topic name invalid utf8"
142 );
143 }
144
145 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
146 for (auto e : rb) {
147 remaining_length_buf_.push_back(e);
148 }
149 switch (pubopts.get_qos()) {
150 case qos::at_most_once:
151 if (packet_id != 0) {
152 throw make_error(
153 errc::bad_message,
154 "v3_1_1::publish_packet qos0 but non 0 packet_id"
155 );
156 }
157 endian_store(0, packet_id_.data());
158 break;
159 case qos::at_least_once:
160 case qos::exactly_once:
161 if (packet_id == 0) {
162 throw make_error(
163 errc::bad_message,
164 "v3_1_1::publish_packet qos not 0 but packet_id is 0"
165 );
166 }
167 endian_store(packet_id, packet_id_.data());
168 break;
169 default:
170 throw make_error(
171 errc::bad_message,
172 "v3_1_1::publish_packet qos is invalid"
173 );
174 break;
175 }
176 }

350 // private constructor for internal use
351 basic_publish_packet(buffer buf)
352 : packet_id_(PacketIdBytes) {
353 // fixed_header
354 if (buf.empty()) {
355 throw make_error(
356 errc::bad_message,
357 "v3_1_1::publish_packet fixed_header doesn't exist"
358 );
359 }
360 fixed_header_ = static_cast<std::uint8_t>(buf.front());
361 auto qos_value = pub::get_qos(fixed_header_);
362 buf.remove_prefix(1);
363 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
364 if (!cpt_opt || *cpt_opt != control_packet_type::publish) {
365 throw make_error(
366 errc::bad_message,
367 "v3_1_1::publish_packet fixed_header is invalid"
368 );
369 }
370
371 // remaining_length
372 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
373 remaining_length_ = *vl_opt;
374 }
375 else {
376 throw make_error(errc::bad_message, "v3_1_1::publish_packet remaining length is invalid");
377 }
378 if (remaining_length_ != buf.size()) {
379 throw make_error(errc::bad_message, "v3_1_1::publish_packet remaining length doesn't match buf.size()");
380 }
381
382 // topic_name_length
383 if (!insert_advance(buf, topic_name_length_buf_)) {
384 throw make_error(
385 errc::bad_message,
386 "v3_1_1::publish_packet length of topic_name is invalid"
387 );
388 }
389 auto topic_name_length = endian_load<std::uint16_t>(topic_name_length_buf_.data());
390
391 // topic_name
392 if (buf.size() < topic_name_length) {
393 throw make_error(
394 errc::bad_message,
395 "v3_1_1::publish_packet topic_name doesn't match its length"
396 );
397 }
398 topic_name_ = buf.substr(0, topic_name_length);
399
400 if (!utf8string_check(topic_name_)) {
401 throw make_error(
402 errc::bad_message,
403 "v3_1_1::publish_packet topic name invalid utf8"
404 );
405 }
406
407 buf.remove_prefix(topic_name_length);
408
409 // packet_id
410 switch (qos_value) {
411 case qos::at_most_once:
412 endian_store(packet_id_t{0}, packet_id_.data());
413 break;
414 case qos::at_least_once:
415 case qos::exactly_once:
416 if (!copy_advance(buf, packet_id_)) {
417 throw make_error(
418 errc::bad_message,
419 "v3_1_1::publish_packet packet_id doesn't exist"
420 );
421 }
422 break;
423 default:
424 throw make_error(
425 errc::bad_message,
426 "v3_1_1::publish_packet qos is invalid"
427 );
428 break;
429 };
430
431 // payload
432 if (!buf.empty()) {
433 payloads_.emplace_back(force_move(buf));
434 }
435 }
