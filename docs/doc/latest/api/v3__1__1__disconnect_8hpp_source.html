107 // private constructor for internal use
108 disconnect_packet(buffer buf) {
109 // fixed_header
110 if (buf.empty()) {
111 throw make_error(
112 errc::bad_message,
113 "v3_1_1::disconnect_packet fixed_header doesn't exist"
114 );
115 }
116 all_.push_back(buf.front());
117 buf.remove_prefix(1);
118 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(all_.back()));
119 if (!cpt_opt || *cpt_opt != control_packet_type::disconnect) {
120 throw make_error(
121 errc::bad_message,
122 "v3_1_1::disconnect_packet fixed_header is invalid"
123 );
124 }
125
126 // remaining_length
127 if (buf.empty()) {
128 throw make_error(
129 errc::bad_message,
130 "v3_1_1::disconnect_packet remaining_length doesn't exist"
131 );
132 }
133 all_.push_back(buf.front());
134
135 if (static_cast<std::uint8_t>(all_.back()) != 0) {
136 throw make_error(
137 errc::bad_message,
138 "v3_1_1::disconnect_packet remaining_length is invalid"
139 );
140 }
141 }
142
143
144private:
145 static_vector<char, 2> all_;
