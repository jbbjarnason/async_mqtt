50template <std::size_t PacketIdBytes>
51class basic_unsubscribe_packet {
52public:
53 using packet_id_t = typename packet_id_type<PacketIdBytes>::type;
54
61 basic_unsubscribe_packet(
62 packet_id_t packet_id,
63 std::vector<topic_sharename> params
64 )
65 : fixed_header_{make_fixed_header(control_packet_type::unsubscribe, 0b0010)},
66 entries_{force_move(params)},
67 remaining_length_{PacketIdBytes}
68 {
69 topic_length_buf_entries_.reserve(entries_.size());
70 for (auto const& e : entries_) {
71 topic_length_buf_entries_.push_back(
72 endian_static_vector(
73 boost::numeric_cast<std::uint16_t>(e.all_topic().size())
74 )
75 );
76 }
77
78 endian_store(packet_id, packet_id_.data());
79
80 // Check for errors before allocating.
81 for (auto const& e : entries_) {
82 auto size = e.all_topic().size();
83 if (size > 0xffff) {
84 throw make_error(
85 errc::bad_message,
86 "v3_1_1::unsubscribe_packet length of topic is invalid"
87 );
88 }
89 remaining_length_ +=
90 2 + // topic filter length
91 size; // topic filter
92
93 if (!utf8string_check(e.all_topic())) {
94 throw make_error(
95 errc::bad_message,
96 "v3_1_1::unsubscribe_packet topic filter invalid utf8"
97 );
98 }
99 }
100
101 remaining_length_buf_ = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
102 }
103
108 constexpr control_packet_type type() const {
109 return control_packet_type::unsubscribe;
110 }

117 std::vector<as::const_buffer> const_buffer_sequence() const {
118 std::vector<as::const_buffer> ret;
119 ret.reserve(num_of_const_buffer_sequence());
120
121 ret.emplace_back(as::buffer(&fixed_header_, 1));
122
123 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
124
125 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
126
127 BOOST_ASSERT(entries_.size() == topic_length_buf_entries_.size());
128 auto it = topic_length_buf_entries_.begin();
129 for (auto const& e : entries_) {
130 ret.emplace_back(as::buffer(it->data(), it->size()));
131 ret.emplace_back(as::buffer(e.all_topic()));
132 ++it;
133 }
134
135 return ret;
136 }

188 // private constructor for internal use
189 basic_unsubscribe_packet(buffer buf) {
190 // fixed_header
191 if (buf.empty()) {
192 throw make_error(
193 errc::bad_message,
194 "v3_1_1::unsubscribe_packet fixed_header doesn't exist"
195 );
196 }
197 fixed_header_ = static_cast<std::uint8_t>(buf.front());
198 buf.remove_prefix(1);
199 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
200 if (!cpt_opt || *cpt_opt != control_packet_type::unsubscribe) {
201 throw make_error(
202 errc::bad_message,
203 "v3_1_1::unsubscribe_packet fixed_header is invalid"
204 );
205 }
206
207 // remaining_length
208 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
209 remaining_length_ = *vl_opt;
210 }
211 else {
212 throw make_error(errc::bad_message, "v3_1_1::unsubscribe_packet remaining length is invalid");
213 }
214 if (remaining_length_ != buf.size()) {
215 throw make_error(errc::bad_message, "v3_1_1::unsubscribe_packet remaining length doesn't match buf.size()");
216 }
217
218 // packet_id
219 if (!copy_advance(buf, packet_id_)) {
220 throw make_error(
221 errc::bad_message,
222 "v3_1_1::unsubscribe_packet packet_id doesn't exist"
223 );
224 }
225
226 if (remaining_length_ == 0) {
227 throw make_error(errc::bad_message, "v3_1_1::unsubscribe_packet doesn't have entries");
228 }
229
230 while (!buf.empty()) {
231 // topic_length
232 static_vector<char, 2> topic_length_buf;
233 if (!insert_advance(buf, topic_length_buf)) {
234 throw make_error(
235 errc::bad_message,
236 "v3_1_1::unsubscribe_packet length of topic is invalid"
237 );
238 }
239 auto topic_length = endian_load<std::uint16_t>(topic_length_buf.data());
240 topic_length_buf_entries_.push_back(topic_length_buf);
241
242 // topic
243 if (buf.size() < topic_length) {
244 throw make_error(
245 errc::bad_message,
246 "v3_1_1::unsubscribe_packet topic doesn't match its length"
247 );
248 }
249 auto topic = buf.substr(0, topic_length);
250 if (!utf8string_check(topic)) {
251 throw make_error(
252 errc::bad_message,
253 "v3_1_1::unsubscribe_packet topic filter invalid utf8"
254 );
255 }
256 entries_.emplace_back(std::string{topic});
257 buf.remove_prefix(topic_length);
258 }
259 }
260
261private:
262 std::uint8_t fixed_header_;
263 std::vector<static_vector<char, 2>> topic_length_buf_entries_;
264 std::vector<topic_sharename> entries_;
265 static_vector<char, PacketIdBytes> packet_id_ = static_vector<char, PacketIdBytes>(PacketIdBytes);
266 std::size_t remaining_length_;
267 static_vector<char, 4> remaining_length_buf_;
268};
