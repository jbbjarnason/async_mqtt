72 static inline std::optional<topic_alias_type> get_topic_alias(properties const& props) {
73 std::optional<topic_alias_type> ta_opt;
74 for (auto const& prop : props) {
75 prop.visit(
76 overload {
77 [&](property::topic_alias const& p) {
78 ta_opt.emplace(p.val());
79 },
80 [](auto const&) {
81 }
82 }
83 );
84 if (ta_opt) return ta_opt;
85 }
86 return ta_opt;
