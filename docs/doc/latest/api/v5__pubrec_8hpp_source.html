122 std::vector<as::const_buffer> const_buffer_sequence() const {
123 std::vector<as::const_buffer> ret;
124 ret.reserve(num_of_const_buffer_sequence());
125 ret.emplace_back(as::buffer(&fixed_header_, 1));
126 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
127
128 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
129
130 if (reason_code_) {
131 ret.emplace_back(as::buffer(&*reason_code_, 1));
132
133 if (property_length_buf_.size() != 0) {
134 ret.emplace_back(as::buffer(property_length_buf_.data(), property_length_buf_.size()));
135 auto props_cbs = async_mqtt::const_buffer_sequence(props_);
136 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));
137 }
138 }
139
140 return ret;
141 }

158 constexpr std::size_t num_of_const_buffer_sequence() const {
159 return
160 1 + // fixed header
161 1 + // remaining length
162 1 + // packet_id
163 [&] () -> std::size_t {
164 if (reason_code_) {
165 return
166 1 + // reason_code
167 [&] () -> std::size_t {
168 if (property_length_buf_.size() == 0) return 0;
169 return
170 1 + // property length
171 async_mqtt::num_of_const_buffer_sequence(props_);
172 }();
173 }
174 return 0;
175 }();
176 }

224private:
225 basic_pubrec_packet(
226 packet_id_t packet_id,
227 std::optional<pubrec_reason_code> reason_code,
228 properties props
229 )
230 : fixed_header_{
231 make_fixed_header(control_packet_type::pubrec, 0b0000)
232 },
233 remaining_length_{
234 PacketIdBytes
235 },
236 packet_id_(packet_id_.capacity()),
237 reason_code_{reason_code},
238 property_length_(async_mqtt::size(props)),
239 props_(force_move(props))
240 {
241 using namespace std::literals;
242 endian_store(packet_id, packet_id_.data());
243
244 auto guard = unique_scope_guard(
245 [&] {
246 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
247 for (auto e : rb) {
248 remaining_length_buf_.push_back(e);
249 }
250 }
251 );
252
253 if (!reason_code_) return;
254 remaining_length_ += 1;
255
256 if (property_length_ == 0) return;
257
258 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
259 for (auto e : pb) {
260 property_length_buf_.push_back(e);
261 }
262
263 for (auto const& prop : props_) {
264 auto id = prop.id();
265 if (!validate_property(property_location::pubrec, id)) {
266 throw make_error(
267 errc::bad_message,
268 "v5::pubrec_packet property "s + id_to_str(id) + " is not allowed"
269 );
270 }
271 }
272
273 remaining_length_ += property_length_buf_.size() + property_length_;
274 }

290 // private constructor for internal use
291 basic_pubrec_packet(buffer buf) {
292 // fixed_header
293 if (buf.empty()) {
294 throw make_error(
295 errc::bad_message,
296 "v5::pubrec_packet fixed_header doesn't exist"
297 );
298 }
299 fixed_header_ = static_cast<std::uint8_t>(buf.front());
300 buf.remove_prefix(1);
301
302 // remaining_length
303 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
304 remaining_length_ = *vl_opt;
305 }
306 else {
307 throw make_error(errc::bad_message, "v5::pubrec_packet remaining length is invalid");
308 }
309
310 // packet_id
311 if (!insert_advance(buf, packet_id_)) {
312 throw make_error(
313 errc::bad_message,
314 "v5::pubrec_packet packet_id doesn't exist"
315 );
316 }
317
318 if (remaining_length_ == PacketIdBytes) {
319 if (!buf.empty()) {
320 throw make_error(errc::bad_message, "v5::pubrec_packet remaining length is invalid");
321 }
322 return;
323 }
324
325 // reason_code
326 reason_code_.emplace(static_cast<pubrec_reason_code>(buf.front()));
327 buf.remove_prefix(1);
328 switch (*reason_code_) {
329 case pubrec_reason_code::success:
330 case pubrec_reason_code::no_matching_subscribers:
331 case pubrec_reason_code::unspecified_error:
332 case pubrec_reason_code::implementation_specific_error:
333 case pubrec_reason_code::not_authorized:
334 case pubrec_reason_code::topic_name_invalid:
335 case pubrec_reason_code::packet_identifier_in_use:
336 case pubrec_reason_code::quota_exceeded:
337 case pubrec_reason_code::payload_format_invalid:
338 break;
339 default:
340 throw make_error(
341 errc::bad_message,
342 "v5::pubrec_packet connect reason_code is invalid"
343 );
344 break;
345 }
346
347 if (remaining_length_ == 3) {
348 if (!buf.empty()) {
349 throw make_error(errc::bad_message, "v5::pubrec_packet remaining length is invalid");
350 }
351 return;
352 }
353
354 // property
355 auto it = buf.begin();
356 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
357 property_length_ = *pl_opt;
358 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
359 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
360 if (buf.size() < property_length_) {
361 throw make_error(
362 errc::bad_message,
363 "v5::pubrec_packet properties_don't match its length"
364 );
365 }
366 auto prop_buf = buf.substr(0, property_length_);
367 props_ = make_properties(prop_buf, property_location::pubrec);
368 buf.remove_prefix(property_length_);
369 }
370 else {
371 throw make_error(
372 errc::bad_message,
373 "v5::pubrec_packet property_length is invalid"
374 );
375 }
376
377 if (!buf.empty()) {
378 throw make_error(
379 errc::bad_message,
380 "v5::pubrec_packet properties don't match its length"
381 );
382 }
383 }
384
385private:
386 std::uint8_t fixed_header_;
387 std::size_t remaining_length_;
388 static_vector<char, 4> remaining_length_buf_;
389 static_vector<char, PacketIdBytes> packet_id_;
390
391 std::optional<pubrec_reason_code> reason_code_;
392
393 std::size_t property_length_ = 0;
394 static_vector<char, 4> property_length_buf_;
395 properties props_;
396};
