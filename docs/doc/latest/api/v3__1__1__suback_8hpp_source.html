153 // private constructor for internal use
154 basic_suback_packet(buffer buf) {
155 // fixed_header
156 if (buf.empty()) {
157 throw make_error(
158 errc::bad_message,
159 "v3_1_1::suback_packet fixed_header doesn't exist"
160 );
161 }
162 fixed_header_ = static_cast<std::uint8_t>(buf.front());
163 buf.remove_prefix(1);
164 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
165 if (!cpt_opt || *cpt_opt != control_packet_type::suback) {
166 throw make_error(
167 errc::bad_message,
168 "v3_1_1::suback_packet fixed_header is invalid"
169 );
170 }
171
172 // remaining_length
173 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
174 remaining_length_ = *vl_opt;
175 }
176 else {
177 throw make_error(errc::bad_message, "v3_1_1::suback_packet remaining length is invalid");
178 }
179 if (remaining_length_ != buf.size()) {
180 throw make_error(errc::bad_message, "v3_1_1::suback_packet remaining length doesn't match buf.size()");
181 }
182
183 // packet_id
184 if (!copy_advance(buf, packet_id_)) {
185 throw make_error(
186 errc::bad_message,
187 "v3_1_1::suback_packet packet_id doesn't exist"
188 );
189 }
190
191 if (remaining_length_ == 0) {
192 throw make_error(errc::bad_message, "v3_1_1::suback_packet doesn't have entries");
193 }
194
195 while (!buf.empty()) {
196 // suback_return_code
197 if (buf.empty()) {
198 throw make_error(
199 errc::bad_message,
200 "v3_1_1::suback_packet suback_return_code doesn't exist"
201 );
202 }
203 auto rc = static_cast<suback_return_code>(buf.front());
204 entries_.emplace_back(rc);
205 buf.remove_prefix(1);
206 }
207 }
208
209private:
210 std::uint8_t fixed_header_;
211 std::vector<suback_return_code> entries_;
212 static_vector<char, PacketIdBytes> packet_id_ = static_vector<char, PacketIdBytes>(PacketIdBytes);
213 std::size_t remaining_length_;
214 static_vector<char, 4> remaining_length_buf_;
215};
