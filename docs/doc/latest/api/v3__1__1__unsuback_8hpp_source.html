123 // private constructor for internal use
124 basic_unsuback_packet(buffer buf) {
125 // fixed_header
126 if (buf.empty()) {
127 throw make_error(
128 errc::bad_message,
129 "v3_1_1::unsuback_packet fixed_header doesn't exist"
130 );
131 }
132 all_.push_back(buf.front());
133 buf.remove_prefix(1);
134 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(all_.back()));
135 if (!cpt_opt || *cpt_opt != control_packet_type::unsuback) {
136 throw make_error(
137 errc::bad_message,
138 "v3_1_1::unsuback_packet fixed_header is invalid"
139 );
140 }
141
142 // remaining_length
143 if (buf.empty()) {
144 throw make_error(
145 errc::bad_message,
146 "v3_1_1::unsuback_packet remaining_length doesn't exist"
147 );
148 }
149 all_.push_back(buf.front());
150 buf.remove_prefix(1);
151 if (static_cast<std::uint8_t>(all_.back()) != PacketIdBytes) {
152 throw make_error(
153 errc::bad_message,
154 "v3_1_1::unsuback_packet remaining_length is invalid"
155 );
156 }
157
158 // variable header
159 if (buf.size() != PacketIdBytes) {
160 throw make_error(
161 errc::bad_message,
162 "v3_1_1::unsuback_packet variable header doesn't match its length"
163 );
164 }
165 std::copy(buf.begin(), buf.end(), std::back_inserter(all_));
166 }
167
168private:
169 static_vector<char, 2 + PacketIdBytes> all_;
170};
