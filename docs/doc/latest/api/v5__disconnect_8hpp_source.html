111 ret.emplace_back(as::buffer(&fixed_header_, 1));

120 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));

213 using namespace std::literals;
214
215 auto guard = unique_scope_guard(
216 [&] {
217 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
218 for (auto e : rb) {
219 remaining_length_buf_.push_back(e);

221 }
222 );
223
224 if (!reason_code_) return;
225 remaining_length_ += 1;
226
227 if (property_length_ == 0) return;
228
229 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
230 for (auto e : pb) {
231 property_length_buf_.push_back(e);
232 }
233
234 for (auto const& prop : props_) {
235 auto id = prop.id();
236 if (!validate_property(property_location::disconnect, id)) {
237 throw make_error(
238 errc::bad_message,
239 "v5::disconnect_packet property "s + id_to_str(id) + " is not allowed"
240 );
241 }
242 }
243
244 remaining_length_ += property_length_buf_.size() + property_length_;
245 }

260 // private constructor for internal use
261 disconnect_packet(buffer buf) {
262 // fixed_header
263 if (buf.empty()) {
264 throw make_error(
265 errc::bad_message,
266 "v5::disconnect_packet fixed_header doesn't exist"
267 );
268 }
269 fixed_header_ = static_cast<std::uint8_t>(buf.front());
270 buf.remove_prefix(1);
271
272 // remaining_length
273 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
274 remaining_length_ = *vl_opt;
275 }
276 else {
277 throw make_error(errc::bad_message, "v5::disconnect_packet remaining length is invalid");
278 }
279
280 if (remaining_length_ == 0) {
281 if (!buf.empty()) {
282 throw make_error(errc::bad_message, "v5::disconnect_packet remaining length is invalid");
283 }
284 return;
285 }
286
287 // reason_code
288 reason_code_.emplace(static_cast<disconnect_reason_code>(buf.front()));
289 buf.remove_prefix(1);
290 switch (*reason_code_) {
291 case disconnect_reason_code::normal_disconnection:
292 case disconnect_reason_code::disconnect_with_will_message:
293 case disconnect_reason_code::unspecified_error:
294 case disconnect_reason_code::malformed_packet:
295 case disconnect_reason_code::protocol_error:
296 case disconnect_reason_code::implementation_specific_error:
297 case disconnect_reason_code::not_authorized:
298 case disconnect_reason_code::server_busy:
299 case disconnect_reason_code::server_shutting_down:
300 case disconnect_reason_code::keep_alive_timeout:
301 case disconnect_reason_code::session_taken_over:
302 case disconnect_reason_code::topic_filter_invalid:
303 case disconnect_reason_code::topic_name_invalid:
304 case disconnect_reason_code::receive_maximum_exceeded:
305 case disconnect_reason_code::topic_alias_invalid:
306 case disconnect_reason_code::packet_too_large:
307 case disconnect_reason_code::message_rate_too_high:
308 case disconnect_reason_code::quota_exceeded:
309 case disconnect_reason_code::administrative_action:
310 case disconnect_reason_code::payload_format_invalid:
311 case disconnect_reason_code::retain_not_supported:
312 case disconnect_reason_code::qos_not_supported:
313 case disconnect_reason_code::use_another_server:
314 case disconnect_reason_code::server_moved:
315 case disconnect_reason_code::shared_subscriptions_not_supported:
316 case disconnect_reason_code::connection_rate_exceeded:
317 case disconnect_reason_code::maximum_connect_time:
318 case disconnect_reason_code::subscription_identifiers_not_supported:
319 case disconnect_reason_code::wildcard_subscriptions_not_supported:
320 break;
321 default:
322 throw make_error(
323 errc::bad_message,
324 "v5::disconnect_packet connect reason_code is invalid"
325 );
326 break;
327 }
328
329 if (remaining_length_ == 1) {
330 if (!buf.empty()) {
331 throw make_error(errc::bad_message, "v5::disconnect_packet remaining length is invalid");
332 }
333 return;
334 }
335
336 // property
337 auto it = buf.begin();
338 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
339 property_length_ = *pl_opt;
340 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
341 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
342 if (buf.size() < property_length_) {
343 throw make_error(
344 errc::bad_message,
345 "v5::disconnect_packet properties_don't match its length"
346 );
347 }
348 auto prop_buf = buf.substr(0, property_length_);
349 props_ = make_properties(prop_buf, property_location::disconnect);
350 buf.remove_prefix(property_length_);
351 }
352 else {
353 throw make_error(
354 errc::bad_message,
355 "v5::disconnect_packet property_length is invalid"
356 );
357 }
358
359 if (!buf.empty()) {
360 throw make_error(
361 errc::bad_message,
362 "v5::disconnect_packet properties don't match its length"
363 );
364 }
365 }
366
367private:
368 std::uint8_t fixed_header_;
369 std::size_t remaining_length_;
370 static_vector<char, 4> remaining_length_buf_;
371
372 std::optional<disconnect_reason_code> reason_code_;
373
374 std::size_t property_length_ = 0;
375 static_vector<char, 4> property_length_buf_;
376 properties props_;
377};
