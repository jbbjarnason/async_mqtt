36template <std::size_t PacketIdBytes>
37class basic_subscribe_packet {
38public:
39 using packet_id_t = typename packet_id_type<PacketIdBytes>::type;

47 basic_subscribe_packet(
48 packet_id_t packet_id,
49 std::vector<topic_subopts> params
50 )
51 : fixed_header_{make_fixed_header(control_packet_type::subscribe, 0b0010)},
52 entries_{force_move(params)},
53 remaining_length_{PacketIdBytes}
54 {
55 topic_length_buf_entries_.reserve(entries_.size());
56 for (auto const& e : entries_) {
57 topic_length_buf_entries_.push_back(
58 endian_static_vector(
59 boost::numeric_cast<std::uint16_t>(e.all_topic().size())
60 )
61 );
62 }
63
64 endian_store(packet_id, packet_id_.data());
65
66 for (auto const& e : entries_) {
67 // reserved bits check
68 if (static_cast<std::uint8_t>(e.opts()) & 0b11111100) {
69 throw make_error(
70 errc::bad_message,
71 "v3_1_1::subscribe_packet subopts is invalid"
72 );
73 }
74 switch (e.opts().get_qos()) {
75 case qos::at_most_once:
76 case qos::at_least_once:
77 case qos::exactly_once:
78 break;
79 default:
80 throw make_error(
81 errc::bad_message,
82 "v3_1_1::subscribe_packet qos is invalid"
83 );
84 break;
85 }
86
87 auto size = e.all_topic().size();
88 if (size > 0xffff) {
89 throw make_error(
90 errc::bad_message,
91 "v3_1_1::subscribe_packet length of topic is invalid"
92 );
93 }
94 remaining_length_ +=
95 2 + // topic filter length
96 size + // topic filter
97 1; // opts
98
99#if 0 // TBD
100 utf8string_check(e.all_topic());
101#endif
102 }
103
104 remaining_length_buf_ = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
105 }

107 basic_subscribe_packet(buffer buf) {
108 // fixed_header
109 if (buf.empty()) {
110 throw make_error(
111 errc::bad_message,
112 "v3_1_1::subscribe_packet fixed_header doesn't exist"
113 );
114 }
115 fixed_header_ = static_cast<std::uint8_t>(buf.front());
116 buf.remove_prefix(1);
117 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
118 if (!cpt_opt || *cpt_opt != control_packet_type::subscribe) {
119 throw make_error(
120 errc::bad_message,
121 "v3_1_1::subscribe_packet fixed_header is invalid"
122 );
123 }
124
125 // remaining_length
126 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
127 remaining_length_ = *vl_opt;
128 }
129 else {
130 throw make_error(errc::bad_message, "v3_1_1::subscribe_packet remaining length is invalid");
131 }
132 if (remaining_length_ != buf.size()) {
133 throw make_error(errc::bad_message, "v3_1_1::subscribe_packet remaining length doesn't match buf.size()");
134 }
135
136 // packet_id
137 if (!copy_advance(buf, packet_id_)) {
138 throw make_error(
139 errc::bad_message,
140 "v3_1_1::subscribe_packet packet_id doesn't exist"
141 );
142 }
143
144 if (remaining_length_ == 0) {
145 throw make_error(errc::bad_message, "v3_1_1::subscribe_packet doesn't have entries");
146 }
147
148 while (!buf.empty()) {
149 // topic_length
150 static_vector<char, 2> topic_length_buf;
151 if (!insert_advance(buf, topic_length_buf)) {
152 throw make_error(
153 errc::bad_message,
154 "v3_1_1::subscribe_packet length of topic is invalid"
155 );
156 }
157 auto topic_length = endian_load<std::uint16_t>(topic_length_buf.data());
158 topic_length_buf_entries_.push_back(topic_length_buf);
159
160 // topic
161 if (buf.size() < topic_length) {
162 throw make_error(
163 errc::bad_message,
164 "v3_1_1::subscribe_packet topic doesn't match its length"
165 );
166 }
167 auto topic = buf.substr(0, topic_length);
168 buf.remove_prefix(topic_length);
169
170 // opts
171 if (buf.empty()) {
172 throw make_error(
173 errc::bad_message,
174 "v3_1_1::subscribe_packet subscribe options doesn't exist"
175 );
176 }
177 auto opts = static_cast<sub::opts>(std::uint8_t(buf.front()));
178 if (static_cast<std::uint8_t>(opts) & 0b11111100) {
179 throw make_error(
180 errc::bad_message,
181 "v3_1_1::subscribe_packet subopts is invalid"
182 );
183 }
184 switch (opts.get_qos()) {
185 case qos::at_most_once:
186 case qos::at_least_once:
187 case qos::exactly_once:
188 break;
189 default:
190 throw make_error(
191 errc::bad_message,
192 "v3_1_1::subscribe_packet qos is invalid"
193 );
194 break;
195 }
196 entries_.emplace_back(force_move(topic), opts);
197 buf.remove_prefix(1);
198 }
199 }

210 std::vector<as::const_buffer> const_buffer_sequence() const {
211 std::vector<as::const_buffer> ret;
212 ret.reserve(num_of_const_buffer_sequence());
213
214 ret.emplace_back(as::buffer(&fixed_header_, 1));
215
216 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
217
218 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
219
220 BOOST_ASSERT(entries_.size() == topic_length_buf_entries_.size());
221 auto it = topic_length_buf_entries_.begin();
222 for (auto const& e : entries_) {
223 ret.emplace_back(as::buffer(it->data(), it->size()));
224 ret.emplace_back(as::buffer(e.all_topic()));
225 ret.emplace_back(as::buffer(&e.opts(), 1));
226 ++it;
227 }
228
229 return ret;
230 }

236 std::size_t size() const {

247 std::size_t num_of_const_buffer_sequence() const {

253 }

259 packet_id_t packet_id() const {

269 }

271private:

273 std::vector<static_vector<char, 2>> topic_length_buf_entries_;
274 std::vector<topic_subopts> entries_;
275 static_vector<char, PacketIdBytes> packet_id_ = static_vector<char, PacketIdBytes>(PacketIdBytes);

277 static_vector<char, 4> remaining_length_buf_;
