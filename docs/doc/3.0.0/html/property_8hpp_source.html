68template <std::size_t N>
69struct n_bytes_property : private boost::totally_ordered<n_bytes_property<N>> {
70 n_bytes_property(property::id id, static_vector<char, N> const& buf)
71 : id_{id},
72 buf_{buf} // very small size copy
73 {
74 }
75
76 template <typename It, typename End>
77 n_bytes_property(property::id id, It b, End e)
78 :id_{id}, buf_(b, e) {}
79
80 n_bytes_property(property::id id, buffer const& buf)
81 :id_{id} {
82 BOOST_ASSERT(buf.size() >= N);
83 buf_.insert(buf_.end(), (buf.begin(), std::next(buf.begin(), N)));
84 }
85
90 std::vector<as::const_buffer> const_buffer_sequence() const {
91 std::vector<as::const_buffer> v;
92 v.reserve(num_of_const_buffer_sequence());
93 v.emplace_back(as::buffer(&id_, 1));
94 v.emplace_back(as::buffer(buf_.data(), buf_.size()));
95 return v;
96 }
97
102 property::id id() const {
103 return id_;
104 }
105
110 std::size_t size() const {
111 return 1 + buf_.size();
112 }
113
118 static constexpr std::size_t num_of_const_buffer_sequence() {
119 return 2;
120 }
121
122 friend bool operator<(n_bytes_property<N> const& lhs, n_bytes_property<N> const& rhs) {
123 return std::tie(lhs.id_, lhs.buf_) < std::tie(rhs.id_, rhs.buf_);
124 }
125
126 friend bool operator==(n_bytes_property<N> const& lhs, n_bytes_property<N> const& rhs) {
127 return std::tie(lhs.id_, lhs.buf_) == std::tie(rhs.id_, rhs.buf_);
128 }
129
130 static constexpr ostream_format const of_ = ostream_format::direct;
131 property::id id_;
132 static_vector<char, N> buf_;
133};

1102template <typename Property>
1103std::enable_if_t< Property::of_ == detail::ostream_format::binary_string, std::ostream& >
1104operator<<(std::ostream& o, Property const& p) {
1105 // Note this only compiles because both strings below are the same length.
1106 o <<
1107 "{" <<
1108 "id:" << p.id() << "," <<
1109 "val:" <<
1110 [&] {
1111 if (p.val() == payload_format::binary) return "binary";
